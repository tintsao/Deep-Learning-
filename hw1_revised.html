def seed_everything(seed):
    # Set Python random seed
    random.seed(seed)
    
    # Set NumPy random seed
    np.random.seed(seed)
    
    # Set PyTorch random seed for CPU and GPU
    torch.manual_seed(seed)
    if torch.cuda.is_available():
        torch.cuda.manual_seed_all(seed)
    
    # Set PyTorch deterministic operations for cudnn backend
    torch.backends.cudnn.deterministic = True  #True: 
    torch.backends.cudnn.benchmark = False     #True: False

class HandwrittenDigits(torch.utils.data.Dataset):
    def __init__(self, df, mode=None, transforms=None):
        self.df = df
        self.transforms = transforms 
        self.mode = mode
        
    def __len__(self):
        return len(self.df)
        
    def __getitem__(self, x):
        if self.mode == "train":
            ID = self.df.iloc[x, 0]                                 # x  0  image_id
            label = self.df.iloc[x, 1]                              # x  1  image_label
            image = self.df.iloc[x, 2:].to_numpy(dtype=np.float32)  # x  2  image  pixel

            if self.transforms:
                image = image.reshape(16, 16)                    # 
                transformed = self.transforms(image=image)
                image = transformed["image"].view(-1)            #  (256,)  MLP 
                
            else:
                image = torch.tensor(image).view(-1)             #  transform  tensor
                label = torch.tensor(label)                          # label  tensorint
                
            # label, image = torch.tensor(label), torch.tensor(image)
            
            # if self.transforms: # train_data  valid_data 
            #     transformed = self.transforms(image=image)
            #     image = transformed['image']
                
            return ID, label, image
        
        elif self.mode == "valid":
            ID = self.df.iloc[x, 0]                                 # x  0  image_id
            label = self.df.iloc[x, 1]                              # x  1  image_label
            image = self.df.iloc[x, 2:].to_numpy(dtype=np.float32)  # x  2  image  pixel
            label, image = torch.tensor(label), torch.tensor(image)
            
            return ID, label, image
        
        elif self.mode == "test":
            ID = self.df.iloc[x, 0]
            image = self.df.iloc[x, 1:].to_numpy(dtype=np.float32) 
            image = torch.tensor(image)
            
            return ID, image  #  test  label

class SimpleNN(nn.Module):
    def __init__(self, input_dim, output_dim):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(input_dim, 128)
        self.bn1 = nn.BatchNorm1d(128)
        self.relu1 = nn.ReLU()
        self.dropout1 = nn.Dropout(0.2)

        self.fc2 = nn.Linear(128, 64)
        self.bn2 = nn.BatchNorm1d(64)
        self.relu2 = nn.ReLU()
        self.dropout2 = nn.Dropout(0.2)

        self.final_fc = nn.Linear(64, output_dim)

    def forward(self, x):
        x = self.dropout1(self.relu1(self.bn1(self.fc1(x))))
        x = self.dropout2(self.relu2(self.bn2(self.fc2(x))))
        x = self.final_fc(x)
        return x

def evaluator(preds, gts):
    preds = preds.cpu().numpy() if isinstance(preds, torch.Tensor) else preds #  Tensor  numpy  Tensor 
    gts = gts.cpu().numpy() if isinstance(gts, torch.Tensor) else gts
    
    acc = accuracy_score(preds, gts)             # 
    f1 = f1_score(preds, gts, average="macro")   # F1 score 0~1  model 
    misclassification_rate = 1 - acc             # 
    
    return acc, f1, misclassification_rate

def train_one_epoch(model, train_loader, optimizer, scheduler, criterion, device):
    model.train()                                            # PyTorch  model  dropout  batch normalization
    train_loss = .0                                          #  epoch 
    predictions, ground_truths = [], []                      # 
    for ID, labels, images in train_loader:
        labels = labels.to(device=device, dtype=torch.long)  # nn.CrossEntropyLoss  label  int 
        images = images.to(device=device, dtype=torch.float) #  image  GPU float
        
        optimizer.zero_grad()                                #  optimizer 
        logits = model(images)                               #  image  model  logits
        loss = criterion(logits, labels)                     #  model loss
        loss.backward()                                      #  loss  model 
        
        optimizer.step()                                     #  model 
        
        train_loss += loss.item()                            #  batch 
        preds = torch.argmax(logits, dim=1)                  # 
        
        predictions.append(preds)                            # 
        ground_truths.append(labels)                         # 
        
    # scheduler.step()                                         #  lr 
    
    train_loss /= len(train_loader)                          #  epoch 
    
    predictions = torch.cat(predictions)                     #  batch 
    ground_truths = torch.cat(ground_truths)                 # 
    
    train_acc, train_f1, train_misclass = evaluator(predictions, ground_truths) #  epoch  acc  F1 score  misclassification_rate
    
    return train_loss, 100*train_acc, 100*train_f1, 100*train_misclass

def validation(model, valid_loader, criterion, device, scheduler):
    model.eval()
    valid_loss = .0
    predictions, ground_truths = [], []
    with torch.no_grad(): # 
        for ID, labels, images in valid_loader:
            labels = labels.to(device=device, dtype=torch.long)
            images = images.to(device=device, dtype=torch.float)
            
            # validation  optimizer  loss.backward()  scheduler
            logits = model(images)  # dimension = out_dim = 10
            loss = criterion(logits, labels)  
            
            valid_loss += loss.item()
            preds = torch.argmax(logits, dim=1)
        
            predictions.append(preds)
            ground_truths.append(labels)
        

    valid_loss /= len(valid_loader)
    
    predictions = torch.cat(predictions)
    ground_truths = torch.cat(ground_truths)
    
    valid_acc, valid_f1, valid_misclass = evaluator(predictions, ground_truths)
    
    return valid_loss, 100*valid_acc, 100*valid_f1, 100*valid_misclass

def Run():
    
    seed_everything(config.seed)
    device = config.device
    
    train_data = pd.read_csv(os.path.join(config.root, 'train.csv'), header=None)
    
    # Split the training dataset in a 80: 20
    train_dataset, val_dataset = train_test_split(train_data, train_size=0.8, test_size=0.2, random_state=config.seed)
    
    train_dataset = HandwrittenDigits(train_dataset, mode = "train", transforms = transform)
    val_dataset = HandwrittenDigits(val_dataset, mode = "valid")
    
    train_loader = torch.utils.data.DataLoader(train_dataset, batch_size=config.batch_size, shuffle=True)  # 
    val_loader = torch.utils.data.DataLoader(val_dataset, batch_size=config.batch_size, shuffle=False)  # 
    
    criterion = nn.CrossEntropyLoss()
    
    model =  SimpleNN(input_dim=256, output_dim=10)
    model.to(device)
    
    parameters = [p for p in model.parameters() if p.requires_grad]  #  model 
    optimizer = torch.optim.AdamW(parameters, lr=config.lr, weight_decay=config.weight_decay)  
    # scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size = 12, gamma = 0.1)  #  12  epoch 0.1
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer=optimizer, 
        mode='min',              #  val loss
        factor=0.5,              #  0.5
        patience=3,              #  3  lr
        verbose=True             # 
    )
    
    best_val_loss = float("inf")  #  val loss 
    history = {
        "train": {
            "loss": [],
            "acc": [],
            "f1": [],
            "misclass": []
        },
        "valid": {
            "loss": [],
            "acc": [],
            "f1": [],
            "misclass": []
        },
    }

    for epoch in range(config.epochs):

        history["valid"]["acc"].append(valid_acc)
        history["valid"]["f1"].append(valid_f1)
        history["valid"]["misclass"].append(valid_misclass)

        print(f'Epoch[{epoch+1}/{config.epochs}], Train Loss: {train_loss:.4f}, Train Acc: {train_acc:.2f}%, Train F1: {train_f1:.2f}% | Valid Loss: {valid_loss:.4f}, Valid Acc: {valid_acc:.2f}%, Valid F1: {valid_f1:.2f}%, Valid Misclass: {valid_misclass:.2f}% | LR: {optimizer.state_dict()["param_groups"][0]["lr"]:.6f}')
    
        if valid_loss < best_val_loss:
            save_file = {

def testing(model, test_loader, device):
    model.eval()
    all_ids = []
    all_preds = []
    
    with torch.no_grad():
        for ID, images in test_loader:
            images = images.to(device=device, dtype=torch.float)
            logits = model(images)
            preds = torch.argmax(logits, dim=1)
            all_ids.extend(ID)
            all_preds.extend(preds.cpu().numpy())
    
    return all_ids, all_preds
